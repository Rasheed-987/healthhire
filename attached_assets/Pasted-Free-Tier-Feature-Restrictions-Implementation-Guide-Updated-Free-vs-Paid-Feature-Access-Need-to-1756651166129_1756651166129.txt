javascriptconst generateCVPDF = async (profileData) => {
  // Use library like jsPDF or Puppeteer for PDF generation
  const { jsPDF } = require('jspdf');
  const doc = new jsPDF();

  // NHS CV Template Layout
  doc.setFontSize(20);
  doc.setTextColor(0, 94, 184); // NHS Blue
  doc.text(profileData.name, 20, 30);
  
  // Contact Information
  doc.setFontSize(11);
  doc.setTextColor(0, 0, 0);
  doc.text(`${profileData.email} | ${profileData.phone} | ${profileData.location}`, 20, 40);
  
  // Professional Summary Section
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('PROFESSIONAL SUMMARY', 20, 55);
  
  doc.setFontSize(10);
  doc.setTextColor(0, 0, 0);
  const summaryLines = doc.splitTextToSize(profileData.summary, 170);
  doc.text(summaryLines, 20, 65);
  
  // Key Skills Section
  let yPosition = 85;
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('KEY SKILLS & COMPETENCIES', 20, yPosition);
  
  yPosition += 10;
  doc.setFontSize(10);
  doc.setTextColor(0, 0, 0);
  profileData.skills.forEach(skill => {
    doc.text(`â€¢ ${skill}`, 20, yPosition);
    yPosition += 5;
  });
  
  // Work Experience Section
  yPosition += 10;
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('PROFESSIONAL EXPERIENCE', 20, yPosition);
  
  profileData.workExperience.forEach(job => {
    yPosition += 10;
    doc.setFontSize(12);
    doc.setTextColor(0, 0, 0);
    doc.text(`${job.title} - ${job.company}`, 20, yPosition);
    
    yPosition += 5;
    doc.setFontSize(9);
    doc.setTextColor(100, 100, 100);
    doc.text(`${job.startDate} - ${job.endDate} | ${job.location}`, 20, yPosition);
    
    yPosition += 7;
    doc.setFontSize(10);
    doc.setTextColor(0, 0, 0);
    job.duties.forEach(duty => {
      const dutyLines = doc.splitTextToSize(`â€¢ ${duty}`, 160);
      doc.text(dutyLines, 25, yPosition);
      yPosition += dutyLines.length * 4;
    });
    yPosition += 3;
  });
  
  // Education Section
  if (yPosition > 250) {
    doc.addPage();
    yPosition = 20;
  }
  
  yPosition += 10;
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('EDUCATION & QUALIFICATIONS', 20, yPosition);
  
  profileData.education.forEach(edu => {
    yPosition += 10;
    doc.setFontSize(11);
    doc.setTextColor(0, 0, 0);
    doc.text(`${edu.qualification} - ${edu.institution}`, 20, yPosition);
    
    yPosition += 5;
    doc.setFontSize(9);
    doc.setTextColor(100, 100, 100);
    doc.text(`${edu.year} | ${edu.grade}`, 20, yPosition);
  });
  
  // Professional Registrations
  yPosition += 15;
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('PROFESSIONAL REGISTRATIONS', 20, yPosition);
  
  profileData.registrations.forEach(reg => {
    yPosition += 8;
    doc.setFontSize(10);
    doc.setTextColor(0, 0, 0);
    doc.text(`â€¢ ${reg.body}: ${reg.number} (Expires: ${reg.expiry})`, 20, yPosition);
  });
  
  // References
  yPosition += 15;
  doc.setFontSize(14);
  doc.setTextColor(0, 94, 184);
  doc.text('REFERENCES', 20, yPosition);
  
  yPosition += 8;
  doc.setFontSize(10);
  doc.setTextColor(0, 0, 0);
  doc.text('Available upon request', 20, yPosition);
  
  return doc;
};

// CV Download Function
const downloadCVPDF = async (profileData) => {
  try {
    const pdf = await generateCVPDF(profileData);
    pdf.save(`${profileData.name.replace(' ', '_')}_NHS_CV.pdf`);
  } catch (error) {
    console.error('PDF generation failed:', error);
  }
