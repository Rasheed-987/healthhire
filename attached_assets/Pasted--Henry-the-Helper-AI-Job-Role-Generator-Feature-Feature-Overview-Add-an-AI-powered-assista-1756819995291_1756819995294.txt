```jsx

  const [isGenerating, setIsGenerating] = useState(false);

      return;

};
```

### 2. ChatGPT Integration

#### API Call Function
```javascript
const generateJobDutiesWithHenry = async (jobTitle) => {
  const prompt = `Create 4 job duties for role ${jobTitle}, make them human and simple language easy to understand dont use dashes`;
  
  try {
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`
      },
      body: JSON.stringify({
        model: 'gpt-3.5-turbo',
        messages: [
          {
            role: 'system',
            content: 'You are Henry the Helper, a friendly AI assistant that helps people write job duties. Write in simple, human language without using bullet points or dashes. Make each duty 1-2 sentences long and easy to understand.'
          },
          {
            role: 'user',
            content: prompt
          }
        ],
        max_tokens: 500,
        temperature: 0.7
      })
    });
    
    const data = await response.json();
    
    if (data.choices && data.choices[0]) {
      const content = data.choices[0].message.content;
      return parseJobDuties(content);
    }
    
    throw new Error('No response from Henry');
    
  } catch (error) {
    console.error('Henry encountered an error:', error);
    throw new Error('Henry is taking a break. Please try again later!');
  }
};

// Parse the response into individual duties
const parseJobDuties = (content) => {
  // Split by numbers (1., 2., 3., 4.) or by line breaks
  const duties = content
    .split(/\d+\./)
    .map(duty => duty.trim())
    .filter(duty => duty.length > 0)
    .slice(0, 4); // Ensure only 4 duties
    
  return duties;
};
```

### 3. Main Component Integration

```javascript
const JobApplicationForm = () => {
  const [jobTitle, setJobTitle] = useState('');
  const [generatedDuties, setGeneratedDuties] = useState(null);
  const [jobDuties, setJobDuties] = useState(['', '', '', '']);
  
  const handleGenerateWithHenry = async (title) => {
    try {
      const duties = await generateJobDutiesWithHenry(title);
      setGeneratedDuties(duties);
    } catch (error) {
      alert(`Henry says: ${error.message}`);
    }
  };
  
  const handleAcceptDuties = () => {
    setJobDuties(generatedDuties);
    setGeneratedDuties(null);
  };
  
  const handleRegenerateDuties = () => {
    handleGenerateWithHenry(jobTitle);
  };
  
  return (
    <div className="job-form">
      <div className="form-group">
        <label htmlFor="jobTitle">Job Title:</label>
        <input
          type="text"
          value={jobTitle}
          onChange={(e) => setJobTitle(e.target.value)}
          placeholder="e.g., Frontend Developer"
        />
        
        <HenryGenerateButton 
          jobTitle={jobTitle}
          onGenerate={handleGenerateWithHenry}
        />
      </div>
      
      {generatedDuties && (
        <HenryResults
          jobDuties={generatedDuties}
          onAccept={handleAcceptDuties}
          onRegenerate={handleRegenerateDuties}
        />
      )}
      
      {/* Rest of your form... */}
    </div>
  );
};
```

## Error Handling & Edge Cases

1. **Empty Job Title:** Show alert asking user to enter job title first
2. **API Failure:** Display friendly message "Henry is taking a break"
3. **Invalid Response:** Fallback to manual entry with helpful message
4. **Rate Limiting:** Implement cooldown period between requests
5. **Long Job Titles:** Truncate or validate input length

## Security & Cost Considerations

1. **API Key Security:** Store in environment variables, never in frontend
2. **Rate Limiting:** Implement user-based rate limiting
3. **Input Sanitization:** Clean job title input before sending to API
4. **Cost Control:** Monitor API usage and set monthly limits
5. **Caching:** Cache common job titles to reduce API calls
